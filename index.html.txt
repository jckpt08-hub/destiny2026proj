  <script>
    const historyStack = ['front'];
    let currentPageId = 'front';
    let isTransitioning = false;

    function updateBackButton() {
      const backBtn = document.getElementById('backBtn');
      backBtn.style.display = historyStack.length > 1 ? 'block' : 'none';
    }

    function cleanupPages(targetPage) {
      document.querySelectorAll('.page').forEach(page => {
        page.classList.remove('enter-forward', 'exit-forward', 'enter-back', 'exit-back', 'active');
      });
      targetPage.classList.add('active');
    }

    function animateTransition(fromPage, toPage, direction) {
      if (!toPage || isTransitioning) return;
      if (fromPage === toPage) return;

      isTransitioning = true;
      const enterClass = direction === 'back' ? 'enter-back' : 'enter-forward';
      const exitClass = direction === 'back' ? 'exit-back' : 'exit-forward';

      document.querySelectorAll('.page').forEach(page => {
        page.classList.remove('enter-forward', 'exit-forward', 'enter-back', 'exit-back');
      });

      if (fromPage) {
        fromPage.classList.add('active');
        fromPage.classList.add(exitClass);
      }

      toPage.classList.add('active');
      toPage.classList.add(enterClass);

      setTimeout(() => {
        cleanupPages(toPage);
        isTransitioning = false;
      }, 360);
    }

    function showPage(pageId, options = {}) {
      const { push = true, direction = 'forward' } = options;
      const nextPage = document.getElementById(pageId);
      const currentPage = document.getElementById(currentPageId);
      if (!nextPage || pageId === currentPageId || isTransitioning) return;

      if (push) {
        historyStack.push(pageId);
      }

      animateTransition(currentPage, nextPage, direction);
      currentPageId = pageId;
      updateBackButton();
      window.location.hash = pageId;
    }

    function goBack() {
      if (historyStack.length <= 1 || isTransitioning) return;
      historyStack.pop();
      const prevPageId = historyStack[historyStack.length - 1];
      showPage(prevPageId, { push: false, direction: 'back' });
    }

    window.addEventListener('load', () => {
      const backBtn = document.getElementById('backBtn');
      backBtn.addEventListener('click', goBack);

      const hash = window.location.hash.replace('#', '');
      const validPage = hash ? document.getElementById(hash) : null;

      if (validPage) {
        historyStack.length = 0;
        historyStack.push('front', hash);
        currentPageId = hash;
        cleanupPages(validPage);
      } else {
        const frontPage = document.getElementById('front');
        historyStack.length = 0;
        historyStack.push('front');
        currentPageId = 'front';
        cleanupPages(frontPage);
      }

      updateBackButton();
    });
  </script>
